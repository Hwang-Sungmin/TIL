		var disp = '';

			var display = document.getElementById('display');

			display.value = disp;
		};
		var del = function () {
			disp = '';
			display.value = disp;
		};
		var sol = function () {
			var display = document.getElementById('display');

			try {
				var result = eval(display.value);
			}
			catch (error) {
				msg = "error" + error.message;
				alert(msg);
				disp = '';
			}
			disp = '';

			display.value = result;
		};
